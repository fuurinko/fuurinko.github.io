 for ( j = 0; j <= 3; ++j )
  {
    for ( k = 0; k < 4 - j; ++k )
    {
      if ( *((_BYTE *)&v15 + k - 14) > *((_BYTE *)&v15 + k - 13) )
      {
        v4 = *((_BYTE *)&v15 + k - 14);
        *((_BYTE *)&v15 + k - 14) = *((_BYTE *)&v15 + k - 13);
        *((_BYTE *)&v15 + k - 13) = v4;
      }
    }
  }
