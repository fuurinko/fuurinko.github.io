    {
      if ( *((_BYTE *)&v15 + k - 14) > *((_BYTE *)&v15 + k - 13) )
      {
        v4 = *((_BYTE *)&v15 + k - 14);
        *((_BYTE *)&v15 + k - 14) = *((_BYTE *)&v15 + k - 13);
        *((_BYTE *)&v15 + k - 13) = v4;
      }
    }
  }
